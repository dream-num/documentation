## Presets 安装

```typescript {4-7,10-11,19-20,26-27}
import { createUniver, defaultTheme, LocaleType, merge } from '@univerjs/presets';
import { UniverDocsCorePreset } from '@univerjs/presets/preset-docs-core';
import UniverPresetDocsCoreZhCN from '@univerjs/presets/preset-docs-core/locales/zh-CN';
import { UniverDocsAdvancedPreset } from '@univerjs/presets/preset-docs-advanced';
import UniverPresetDocsAdvancedZhCN from '@univerjs/presets/preset-docs-advanced/locales/zh-CN';
import { UniverDocsDrawingPreset } from '@univerjs/presets/preset-docs-drawing'
import UniverPresetDocsDrawingZhCN from '@univerjs/presets/preset-docs-drawing/locales/zh-CN'

import '@univerjs/presets/lib/styles/preset-docs-core.css'
import '@univerjs/presets/lib/styles/preset-docs-drawing.css'
import '@univerjs/presets/lib/styles/preset-docs-advanced.css'

const { univerAPI } = createUniver({
  locale: LocaleType.ZH_CN,
  locales: {
    [LocaleType.ZH_CN]: merge(
      {},
      UniverPresetDocsCoreZhCN,
      UniverPresetDocsDrawingZhCN,
      UniverPresetDocsAdvancedZhCN,
    ),
  },
  theme: defaultTheme,
  presets: [
    UniverDocsCorePreset(),
    UniverDocsDrawingPreset(),
    UniverDocsAdvancedPreset(),
  ],
});
```

## 手动组合安装

```shell npm2yarn
npm install @univerjs-pro/docs-print
```

```typescript {3-4,6,13,18}
import { LocaleType, merge, Univer } from '@univerjs/core';
import { defaultTheme } from "@univerjs/design";
import { UniverDocsPrintPlugin } from '@univerjs-pro/docs-print'
import DocsPrintPluginZhCN from '@univerjs-pro/docs-print/locale/zh-CN';

import '@univerjs-pro/docs-print/lib/index.css';

const univer = new Univer({
  theme: defaultTheme,
  locale: LocaleType.ZH_CN,
  locales: {
    [LocaleType.ZH_CN]: merge(
      DocsPrintPluginZhCN
    ),
  },
});

univer.registerPlugin(UniverDocsPrintPlugin);
```
